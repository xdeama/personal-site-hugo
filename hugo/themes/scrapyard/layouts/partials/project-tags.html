Never rendered on the /portfolio print page; tags.css additionally hides .tag-group under @media print as a safety net. */ -}} {{- $page := . -}} {{- $labels := false -}} {{- if reflect.IsMap . -}} {{- $page = .page -}} {{- $labels = .labels -}} {{- end -}} {{- $tech := $page.GetTerms "technologies" -}} {{- $areas := $page.GetTerms "business-areas" -}} {{- if or $tech $areas }}
{{- with $tech }}
{{- if $labels }}{{ T "technologies" }}:{{ end }}
    {{- range . }}
  • {{ .LinkTitle }}
    • {{- end }}
{{- end }} {{- with $areas }}
{{- if $labels }}{{ if gt (len .) 1 }}{{ T "businessAreas" }}{{ else }}{{ T "businessArea" }}{{ end }}:{{ end }}
    {{- range . }}
  • {{ .LinkTitle }}
    • {{- end }}
{{- end }}
{{- end }}
