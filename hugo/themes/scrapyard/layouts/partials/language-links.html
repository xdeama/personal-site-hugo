{{- /* Renders all site languages as short codes, current one muted and not a link. Falls back to the other language's homepage when the current page has no translation. Context: dict "page" page, optional "icon" bool for the translate glyph (Material Design icons, Apache-2.0). */ -}} {{- $page := .page -}} {{- $current := $page.Language.Lang -}} {{- if .icon -}} {{- end -}} {{- range $i, $s := site.Sites -}} {{- $lang := $s.Language.Lang -}} {{- if gt $i 0 }}|{{ end -}} {{- if eq $lang $current -}} {{ $lang }} {{- else -}} {{- $target := $s.Home.Permalink -}} {{- range $page.Translations }}{{ if eq .Lang $lang }}{{ $target = .Permalink }}{{ end }}{{ end -}} {{ $lang }} {{- end -}} {{- end -}}
